** wgrep
Lets us edit the results of a grep search while inside a grep-mode buffer
#+begin_src emacs-lisp
  (use-package wgrep
    :bind
    (:map grep-mode-map
          ("e" . wgrep-change-to-wgrep-mode)
          :map wgrep-mode-map
          ("C-c C-c" . wgrep-finish-edit)
          ;; Remove default finish bindings so C-c C-c is the only way out
          ("C-x C-s" . nil)
          ("C-c C-e" . nil)))
#+end_src
